37  static T CalculateLerp(const T &Start, const T &End,
38  float StartTime, float EndTime, float NowTime, rate r){
39  float t = (NowTime - StartTime) / (EndTime - StartTime);
40  if (t < 0.0f){
41  t = 0;
42  }
43  if (t > 1.0f){
44  t = 1.0f;
45  }
46  float cal_rate = t; // 1次関数補間値に変換(デフォルト)
47  switch (r){
48  case Linear:
49  cal_rate = t;
50  break;
51  case Easein:
52  cal_rate = t * t;
53  break;
54  case EaseOut:
55  cal_rate = t * (2.0f - t);
56  break;
57  case Cube:
58  cal_rate = t * t * (3.0f - 2.0f * t);
59  break;
60  case Cos:
61  cal_rate = (1.0f - cos(t * XM_PI)) / 2.0f;
62  break;
63  default:
64  break;
65  }
66  return Start * (1.0f - cal_rate) + End * cal_rate;
67  }
