680  template<typename T_VShader, typename T_PShader>
681  void DrawStaticNoLight() {
682  auto PtrStage = GetGameObject()->GetStage();
683  if (!PtrStage) {
684  return;
685  }
686  //メッシュがなければ描画しない
687  auto MeshRes = GetMeshResource();
688  if (!MeshRes) {
689  throw BaseException(
690  L"メッシュが作成されていません",
691  L"if (!MeshRes)",
692  L"BcStaticBaseDraw::DrawStaticNoLight()"
693  );
694  }
695  auto shTex = GetTextureResource();
696  auto Dev = App::GetApp()->GetDeviceResources();
697  auto pD3D11DeviceContext = Dev->GetD3DDeviceContext();
698  auto RenderState = Dev->GetRenderState();
699  ID3D11ShaderResourceView* pNull[D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT] = { nullptr };
700  ID3D11SamplerState* pNullSR[D3D11_COMMONSHADER_SAMPLER_SLOT_COUNT] = { nullptr };
701 
702  //コンスタントバッファの設定
703  BasicConstants BcCb;
704  SetConstants(BcCb);
705  if (shTex) {
706  //テクスチャがある
707  BcCb.activeFlg.y = 1;
708  }
709  else {
710  //テクスチャがない
711  BcCb.activeFlg.y = 0;
712  }
713  //コンスタントバッファの更新
714  pD3D11DeviceContext->UpdateSubresource(CBBasic::GetPtr()->GetBuffer(), 0, nullptr, &BcCb, 0, 0);
715  //コンスタントバッファの設定
716  ID3D11Buffer* pConstantBuffer = CBBasic::GetPtr()->GetBuffer();
717  ID3D11Buffer* pNullConstantBuffer = nullptr;
718  //頂点シェーダに渡す
719  pD3D11DeviceContext->VSSetConstantBuffers(0, 1, &pConstantBuffer);
720  //ピクセルシェーダに渡す
721  pD3D11DeviceContext->PSSetConstantBuffers(0, 1, &pConstantBuffer);
722 
723  //シェーダの設定
724  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
725  //インプットレイアウトの設定
726  pD3D11DeviceContext->IASetInputLayout(T_VShader::GetPtr()->GetInputLayout());
727  //ピクセルシェーダ
728  pD3D11DeviceContext->PSSetShader(T_PShader::GetPtr()->GetShader(), nullptr, 0);
729 
730 
731  //テクスチャとサンプラー
732  if (shTex) {
733  pD3D11DeviceContext->PSSetShaderResources(0, 1, shTex->GetShaderResourceView().GetAddressOf());
734  //サンプラーは設定に任せる
735  SetDeviceSamplerState();
736  }
737  else {
738  //シェーダーリソースもクリア
739  pD3D11DeviceContext->PSSetShaderResources(0, D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT, pNull);
740  //サンプラーもクリア
741  pD3D11DeviceContext->PSSetSamplers(0, D3D11_COMMONSHADER_SAMPLER_SLOT_COUNT, pNullSR);
742  }
743 
744  //ストライドとオフセット
745  UINT stride = MeshRes->GetNumStride();
746  UINT offset = 0;
747  //頂点バッファのセット
748  pD3D11DeviceContext->IASetVertexBuffers(0, 1, MeshRes->GetVertexBuffer().GetAddressOf(), &stride, &offset);
749  //インデックスバッファのセット
750  pD3D11DeviceContext->IASetIndexBuffer(MeshRes->GetIndexBuffer().Get(), DXGI_FORMAT_R16_UINT, 0);
751  //描画方法（3角形）
752  pD3D11DeviceContext->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
753  //デプスステンシルステートは設定に任せる
754  SetDeviceDepthStencilState();
755  //透明処理なら
756  if (GetGameObject()->GetAlphaActive()) {
757  //ブレンドステート
758  //透明処理
759  if (GetBlendState() == BlendState::Additive) {
760  pD3D11DeviceContext->OMSetBlendState(RenderState->GetAdditive(), nullptr, 0xffffffff);
761  }
762  else {
763  pD3D11DeviceContext->OMSetBlendState(RenderState->GetAlphaBlendEx(), nullptr, 0xffffffff);
764  }
765  //ラスタライザステート(裏描画)
766  pD3D11DeviceContext->RSSetState(RenderState->GetCullFront());
767  //描画
768  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
769  //ラスタライザステート（表描画）
770  pD3D11DeviceContext->RSSetState(RenderState->GetCullBack());
771  //描画
772  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
773  }
774  else {
775  //透明処理しない
776  //ブレンドステートは設定に任せる
777  SetDeviceBlendState();
778  //ラスタライザステートは設定に任せる
779  SetDeviceRasterizerState();
780  //描画
781  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
782  }
783  //後始末
784  Dev->InitializeStates();
785  }
786  private:
787  // pImplイディオム
788  struct Impl;
789  unique_ptr<Impl> pImpl;
790  };
