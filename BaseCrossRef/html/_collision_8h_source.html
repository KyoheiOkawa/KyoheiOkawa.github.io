370  template<typename SrcType, typename DestType>
371  void AfterCollisionTmpl_1(const Vector3& SrcVelocity, const Vector3& DestVelocity, float ElapsedTime, float HitTime,
372  const shared_ptr<SrcType>& SrcColl, const shared_ptr<DestType>& DestColl) {
373  if (HitTime <= 0) {
374  HitTime = 0;
375  }
376  if (HitTime >= ElapsedTime) {
377  HitTime = ElapsedTime;
378  }
379  auto AfterHitTime = ElapsedTime - HitTime;
380  bool UseSrcAfter = false;
381  bool UseDestAfter = false;
382  //衝突相手の登録
383  AddHitObject(DestColl->GetGameObject());
384  if (GetIsHitAction() != IsHitAction::None) {
385  //事後処理が有効なら衝突前まで戻る
386  BackToBefore(SrcVelocity, HitTime);
387  UseSrcAfter = true;
388  }
389  if (!DestColl->IsFixed()) {
390  //もし相手がFixでなければ相手の衝突相手の登録
391  DestColl->AddHitObject(GetGameObject());
392  if (DestColl->GetIsHitAction() != IsHitAction::None) {
393  //事後処理が有効なら衝突前まで戻る
394  DestColl->BackToBefore(DestVelocity, HitTime);
395  UseDestAfter = true;
396  }
397  }
398  else {
399  //相手がFixなら自己退避だけでは衝突している可能性があるので判定をしてエスケープ処理
400  if (GetIsHitAction() != IsHitAction::None) {
401  CollisionEscape(DestColl);
402  }
403  }
404  if (UseSrcAfter) {
405  Vector3 HitNormal;
406  GetHitNormal(DestColl, HitNormal);
407  AfterCollision(DestColl, SrcVelocity, DestVelocity, HitNormal, AfterHitTime);
408  }
409  if (UseDestAfter) {
410  Vector3 HitNormal;
411  DestColl->GetHitNormal(SrcColl, HitNormal);
412  DestColl->AfterCollision(SrcColl, DestVelocity, SrcVelocity, HitNormal, AfterHitTime);
413  }
414 
415  }
