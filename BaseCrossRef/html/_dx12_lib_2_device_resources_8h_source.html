164  //デバイスの取得
165  auto Dev = App::GetApp()->GetDeviceResources();
166  auto Ptr = ObjectFactory::Create<MeshResource>();
167  Ptr->m_AccessWrite = AccessWrite;
168  //データのバックアップ
169 
170  auto DataPtr = make_shared< BackupData<T> >();
171  DataPtr->m_Vertices.clear();
172  for (auto v : vertices) {
173  DataPtr->m_Vertices.push_back(v);
174  }
175  DataPtr->m_Indices.clear();
176  for (auto i : indices) {
177  DataPtr->m_Indices.push_back(i);
178  }
179  Ptr->m_BackupData = DataPtr;
180  UINT vertexBufferSize = sizeof(T) * vertices.size();
181  //頂点バッファの作成
182  {
183  ThrowIfFailed(Dev->GetDevice()->CreateCommittedResource(
184  &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
185  D3D12_HEAP_FLAG_NONE,
186  &CD3DX12_RESOURCE_DESC::Buffer(vertexBufferSize),
187  D3D12_RESOURCE_STATE_COPY_DEST,
188  nullptr,
189  IID_PPV_ARGS(&Ptr->m_VertexBuffer)),
190  L"頂点バッファ作成に失敗しました。",
191  L"if(FAILED(pDx12Device->CreateCommittedResource())",
192  L"MeshResource::CreateMeshResource()"
193  );
194 
195  ThrowIfFailed(Dev->GetDevice()->CreateCommittedResource(
196  &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_UPLOAD),
197  D3D12_HEAP_FLAG_NONE,
198  &CD3DX12_RESOURCE_DESC::Buffer(vertexBufferSize),
199  D3D12_RESOURCE_STATE_GENERIC_READ,
200  nullptr,
201  IID_PPV_ARGS(&Ptr->m_VertexBufferUploadHeap)),
202  L"頂点アップロードバッファ作成に失敗しました。",
203  L"if(FAILED(pDx12Device->CreateCommittedResource())",
204  L"MeshResource::CreateMeshResource()"
205  );
206  // Initialize the vertex buffer view.
207  Ptr->m_VertexBufferView.BufferLocation = Ptr->m_VertexBuffer->GetGPUVirtualAddress();
208  Ptr->m_VertexBufferView.StrideInBytes = static_cast<UINT>(sizeof(T));
209  Ptr->m_VertexBufferView.SizeInBytes = static_cast<UINT>(sizeof(T) * vertices.size());
210  }
211  //頂点数の設定
212  Ptr->m_NumVertices = static_cast<UINT>(vertices.size());
213  //インデックスバッファの作成
214  UINT indexBufferSize = static_cast<UINT>(sizeof(uint16_t) * indices.size());
215  {
216  ThrowIfFailed(Dev->GetDevice()->CreateCommittedResource(
217  &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
218  D3D12_HEAP_FLAG_NONE,
219  &CD3DX12_RESOURCE_DESC::Buffer(indexBufferSize),
220  D3D12_RESOURCE_STATE_COPY_DEST,
221  nullptr,
222  IID_PPV_ARGS(&Ptr->m_IndexBuffer)),
223  L"インデックスバッファ作成に失敗しました。",
224  L"if(FAILED(pDx12Device->CreateCommittedResource())",

596  void UpdateResources(const ComPtr<ID3D12GraphicsCommandList>& commandList) {
597  if (!m_DataRefresh) {
598  return;
599  }
600  auto shptr = dynamic_pointer_cast< BackupData<T> >(m_BackupData);
601  if (!shptr) {
602  return;
603  }
604  //頂点バッファの更新
605  D3D12_SUBRESOURCE_DATA vertexData = {};
606  vertexData.pData = (void*)&shptr->m_Vertices[0];
607  vertexData.RowPitch = shptr->m_Vertices.size() * sizeof(T);
608  vertexData.SlicePitch = vertexData.RowPitch;
609  UpdateSubresources<1>(commandList.Get(), m_VertexBuffer.Get(), m_VertexBufferUploadHeap.Get(), 0, 0, 1, &vertexData);
610  commandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(m_VertexBuffer.Get(),
611  D3D12_RESOURCE_STATE_COPY_DEST, D3D12_RESOURCE_STATE_VERTEX_AND_CONSTANT_BUFFER));
612  //インデックスバッファの更新
613  if (shptr->m_Indices.size() > 0) {
614  D3D12_SUBRESOURCE_DATA indexData = {};
615  indexData.pData = (void*)&shptr->m_Indices[0];
616  indexData.RowPitch = shptr->m_Indices.size() * sizeof(uint16_t);
617  indexData.SlicePitch = indexData.RowPitch;
618  UpdateSubresources<1>(commandList.Get(), m_IndexBuffer.Get(), m_IndexBufferUploadHeap.Get(), 0, 0, 1, &indexData);
619  commandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(m_IndexBuffer.Get(), D3D12_RESOURCE_STATE_COPY_DEST, D3D12_RESOURCE_STATE_INDEX_BUFFER));
620  }
621  m_DataRefresh = false;
622  }
