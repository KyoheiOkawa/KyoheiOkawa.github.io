1020  void DrawStatic(bool TextureUse, bool ShadoUuse) {
1021  auto PtrStage = GetGameObject()->GetStage();
1022  if (!PtrStage) {
1023  return;
1024  }
1025  //メッシュがなければ描画しない
1026  auto MeshRes = GetMeshResource();
1027  if (!MeshRes) {
1028  throw BaseException(
1029  L"メッシュが作成されていません",
1030  L"if (!MeshRes)",
1031  L"StaticBaseDraw::DrawStatic()"
1032  );
1033  }
1034  auto Dev = App::GetApp()->GetDeviceResources();
1035  auto pD3D11DeviceContext = Dev->GetD3DDeviceContext();
1036  auto RenderState = Dev->GetRenderState();
1037  //コンスタントバッファの準備
1038  SimpleConstants Cb;
1039  SetConstants(Cb, ShadoUuse);
1040  if (TextureUse) {
1041  //テクスチャ
1042  auto shTex = GetTextureResource();
1043  if (shTex) {
1044  Cb.ActiveFlg.x = 1;
1045  pD3D11DeviceContext->PSSetShaderResources(0, 1, shTex->GetShaderResourceView().GetAddressOf());
1046  //サンプラーは設定に任せる
1047  SetDeviceSamplerState();
1048  }
1049  else {
1050  Cb.ActiveFlg.x = 0;
1051  }
1052  }
1053 
1054  //コンスタントバッファの更新
1055  pD3D11DeviceContext->UpdateSubresource(CBSimple::GetPtr()->GetBuffer(), 0, nullptr, &Cb, 0, 0);
1056  //ストライドとオフセット
1057  UINT stride = MeshRes->GetNumStride();
1058  UINT offset = 0;
1059  //頂点バッファのセット
1060  pD3D11DeviceContext->IASetVertexBuffers(0, 1, MeshRes->GetVertexBuffer().GetAddressOf(), &stride, &offset);
1061  //インデックスバッファのセット
1062  pD3D11DeviceContext->IASetIndexBuffer(MeshRes->GetIndexBuffer().Get(), DXGI_FORMAT_R16_UINT, 0);
1063  //描画方法（3角形）
1064  pD3D11DeviceContext->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
1065 
1066  //コンスタントバッファの設定
1067  ID3D11Buffer* pConstantBuffer = CBSimple::GetPtr()->GetBuffer();
1068  ID3D11Buffer* pNullConstantBuffer = nullptr;
1069  //頂点シェーダに渡す
1070  pD3D11DeviceContext->VSSetConstantBuffers(0, 1, &pConstantBuffer);
1071  //ピクセルシェーダに渡す
1072  pD3D11DeviceContext->PSSetConstantBuffers(0, 1, &pConstantBuffer);
1073  if (ShadoUuse) {
1074  //シェーダの設定
1075  if (GetGameObject()->GetComponent<Shadowmap>(false)) {
1076  //シャドウマップがあれば自己影防止用のピクセルシェーダ
1077  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
1078  pD3D11DeviceContext->PSSetShader(PSPNTStaticShadow2::GetPtr()->GetShader(), nullptr, 0);
1079  }
1080  else {
1081  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
1082  pD3D11DeviceContext->PSSetShader(PSPNTStaticShadow::GetPtr()->GetShader(), nullptr, 0);
1083  }
1084  //シャドウマップのレンダラーターゲット
1085  auto ShadowmapPtr = Dev->GetShadowMapRenderTarget();
1086  ID3D11ShaderResourceView* pShadowSRV = ShadowmapPtr->GetShaderResourceView();
1087  pD3D11DeviceContext->PSSetShaderResources(1, 1, &pShadowSRV);
1088  //シャドウマップサンプラー
1089  ID3D11SamplerState* pShadowSampler = RenderState->GetComparisonLinear();
1090  pD3D11DeviceContext->PSSetSamplers(1, 1, &pShadowSampler);
1091  }
1092  else {
1093  //シェーダの設定
1094  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
1095  pD3D11DeviceContext->PSSetShader(T_PShader::GetPtr()->GetShader(), nullptr, 0);
1096  }
1097  //インプットレイアウトの設定
1098  pD3D11DeviceContext->IASetInputLayout(T_VShader::GetPtr()->GetInputLayout());
1099  //デプスステンシルステートは設定に任せる
1100  SetDeviceDepthStencilState();
1101  //透明処理なら
1102  if (GetGameObject()->GetAlphaActive()) {
1103  //ブレンドステート
1104  //透明処理
1105  if (GetBlendState() == BlendState::Additive) {
1106  pD3D11DeviceContext->OMSetBlendState(RenderState->GetAdditive(), nullptr, 0xffffffff);
1107  }
1108  else {
1109  pD3D11DeviceContext->OMSetBlendState(RenderState->GetAlphaBlendEx(), nullptr, 0xffffffff);
1110  }
1111  //ラスタライザステート(裏描画)
1112  pD3D11DeviceContext->RSSetState(RenderState->GetCullFront());
1113  //描画
1114  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
1115  //ラスタライザステート（表描画）
1116  pD3D11DeviceContext->RSSetState(RenderState->GetCullBack());
1117  //描画
1118  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
1119  }
1120  else {
1121  //透明処理しない
1122  //ブレンドステートは設定に任せる
1123  SetDeviceBlendState();
1124  //ラスタライザステートは設定に任せる
1125  SetDeviceRasterizerState();
1126  //描画
1127  pD3D11DeviceContext->DrawIndexed(MeshRes->GetNumIndicis(), 0, 0);
1128  }
1129  //後始末
1130  Dev->InitializeStates();
1131  }
