320  bool IsMouseEnabled(HWND hWnd, vector<DWORD>& UseKeyVec) {
321  //マウスポイントは0,0に初期化
322  m_MouseClientPoint = { 0,0 };
323  //マウス利用にかかわらず、マウスポインタ座標は設定
324  POINT p;
325  ::GetCursorPos(&p);
326  if (::ScreenToClient(hWnd, &p)) {
327  //クライアント座標に変換できたときのみ、マウス座標を設定
328  m_MouseClientPoint = { p.x, p.y };
329  }
330  else {
331  return false;
332  }
333 
334  //UseKeyVecにマウスがあった場合
335  vector<DWORD> MouseTemp = { VK_LBUTTON, VK_RBUTTON, VK_MBUTTON };
336  bool MouseChk = false;
337  for (auto chk : MouseTemp) {
338  auto it = find(UseKeyVec.begin(), UseKeyVec.end(), chk);
339  if (it != UseKeyVec.end()) {
340  MouseChk = true;
341  break;
342  }
343  }
344  //マウス使用で座標がクライアント領域内ならtrue
345  if (MouseChk) {
346  RECT rc;
347  ::GetClientRect(hWnd, &rc);
348  if (::PtInRect(&rc, p)) {
349  return true;
350  }
351  }
352  return false;

362  bool GetKeyState(HWND hWnd, vector<DWORD>& UseKeyVec) {
363  m_KeyMessageActive = false;
364  //一つ前にコピー
365  CopyMemory(m_bLastKeyTbl, m_bPushKeyTbl, sizeof(m_bLastKeyTbl));
366  //キーボードの状態を初期化
367  ::ZeroMemory(&m_bPushKeyTbl, sizeof(m_bPushKeyTbl));
368  ::ZeroMemory(&m_bPressedKeyTbl, sizeof(m_bPressedKeyTbl));
369  ::ZeroMemory(&m_bUpKeyTbl, sizeof(m_bUpKeyTbl));
370  //マウスのチェック
371  bool MouseEnabled = IsMouseEnabled(hWnd, UseKeyVec);
372  //自分自身にフォーカスがない場合はfalse
373  if (::GetFocus() != hWnd) {
374  return false;
375  }
376  SHORT ret = GetAsyncKeyState((int)VK_ESCAPE);
377  if (ret & 0x8000) {
378  //Escキーは無条件にfalse
379  return false;
380  }
381  size_t sz = UseKeyVec.size();
382  for (size_t i = 0; i < sz; i++) {
383  ret = GetAsyncKeyState((int)UseKeyVec[i]);
384  if (UseKeyVec[i] == VK_LBUTTON || UseKeyVec[i] == VK_RBUTTON || UseKeyVec[i] == VK_MBUTTON) {
385  if (!MouseEnabled) {
386  //マウスが無効ならUsedに入っていても無視
387  continue;
388  }
389  }
390  if (ret & 0x8000) {
391  m_bPushKeyTbl[UseKeyVec[i]] = true;
392  m_KeyMessageActive = true; //メッセージをゲームが受け取る
393  if (!m_bLastKeyTbl[UseKeyVec[i]]) {
394  //前回押されてなくて今回押された
395  m_bPressedKeyTbl[UseKeyVec[i]] = true;
396  }
397  }
398  else {
399  //キーは押されてない
400  if (m_bLastKeyTbl[UseKeyVec[i]]) {
401  //前回押されていて今回押されてない
402  m_bUpKeyTbl[UseKeyVec[i]] = true;
403  m_KeyMessageActive = true; //メッセージをゲームが受け取る
404  }
405  }
406  }
407  return m_KeyMessageActive;
408  }

455  void ResetControlerState() {
456  DWORD dwResult;
457  for (DWORD i = 0; i < MAX_CONTROLLERS; i++)
458  {
459  //一つ前の状態を保存
460  m_State[i].wLastButtons = m_State[i].Gamepad.wButtons;
461  // XInputから現在のステータスを得る
462  XINPUT_STATE State;
463  dwResult = XInputGetState(i, &State);
464  if (dwResult == ERROR_SUCCESS) {
465  m_State[i].dwPacketNumber = State.dwPacketNumber;
466  m_State[i].Gamepad = State.Gamepad;
467  m_State[i].bConnected = true;
468  //ボタンの設定
469  //コピー
470  m_State[i].wButtons = m_State[i].Gamepad.wButtons;
471  //更新ボタン
472  m_State[i].wNowUpdateButtons = m_State[i].Gamepad.wButtons;
473  m_State[i].bLeftTrigger = m_State[i].Gamepad.bLeftTrigger;
474  m_State[i].bRightTrigger = m_State[i].Gamepad.bRightTrigger;
475  //前回より変化しているボタンのみフラグを立てる
476  //XOR演算により実装
477  m_State[i].wNowUpdateButtons ^= m_State[i].wLastButtons;
478  //押された瞬間をセット
479  m_State[i].wPressedButtons = m_State[i].wNowUpdateButtons & m_State[i].wButtons;
480  //離された瞬間をセット
481  m_State[i].wReleasedButtons = m_State[i].wNowUpdateButtons & (m_State[i].wButtons ^ 0xffff);
482  //レバーのデッドゾーンの設定
483  if ((m_State[i].Gamepad.sThumbLX < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
484  m_State[i].Gamepad.sThumbLX > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE) &&
485  (m_State[i].Gamepad.sThumbLY < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
486  m_State[i].Gamepad.sThumbLY > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE))
487  {
488  m_State[i].Gamepad.sThumbLX = 0;
489  m_State[i].Gamepad.sThumbLY = 0;
490  }
491  if (m_State[i].Gamepad.sThumbLX >= 0) {
492  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)SHRT_MAX;
493  }
494  else {
495  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)(SHRT_MAX + 1);
496  }
497  if (m_State[i].Gamepad.sThumbLY >= 0) {
498  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)SHRT_MAX;
499  }
500  else {
501  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)(SHRT_MAX + 1);
502  }
503 
504  if ((m_State[i].Gamepad.sThumbRX < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
505  m_State[i].Gamepad.sThumbRX > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE) &&
506  (m_State[i].Gamepad.sThumbRY < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
507  m_State[i].Gamepad.sThumbRY > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE))
508  {
509  m_State[i].Gamepad.sThumbRX = 0;
510  m_State[i].Gamepad.sThumbRY = 0;
511  }
512  if (m_State[i].Gamepad.sThumbRX >= 0) {
513  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)SHRT_MAX;
514  }
515  else {
516  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)(SHRT_MAX + 1);
517  }
518  if (m_State[i].Gamepad.sThumbRY >= 0) {
519  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)SHRT_MAX;
520  }
521  else {
522  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)(SHRT_MAX + 1);
523  }
524  }
525  else {
526  m_State[i].bConnected = false;
527  }
528  }
