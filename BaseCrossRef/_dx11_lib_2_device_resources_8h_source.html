593  static shared_ptr<MeshResource> CreateMeshResource(const vector<T>& vertices, bool AccessWrite) {
594  std::mutex Mutex;
595  //デバイスの取得
596  auto Dev = App::GetApp()->GetDeviceResources();
597  auto pDx11Device = Dev->GetD3DDevice();
598  auto Ptr = ObjectFactory::Create<MeshResource>();
599  //バッファの作成
600  if (AccessWrite) {
601  Util::DemandCreate(Ptr->m_VertexBuffer, Mutex, [&](ID3D11Buffer** pResult)
602  {
603  auto PtrBackup = shared_ptr< BackupData<T> >(new BackupData<T>());
604  for (auto& v : vertices) {
605  PtrBackup->m_Vertices.push_back(v);
606  }
607  Ptr->m_BackUpData = PtrBackup;
608  //頂点バッファの作成
609  MeshResource::CreateDynamicPrimitiveVertexBuffer(pDx11Device, vertices, pResult);
610  });
611  }
612  else {
613  Util::DemandCreate(Ptr->m_VertexBuffer, Mutex, [&](ID3D11Buffer** pResult)
614  {
615  //頂点バッファの作成
616  MeshResource::CreatePrimitiveBuffer(pDx11Device, vertices, D3D11_BIND_VERTEX_BUFFER, pResult);
617  });
618  }
619  //頂点数の設定
620  Ptr->m_NumVertices = static_cast<UINT>(vertices.size());
621  Ptr->m_MeshTypeIndex = typeid(T);
622  Ptr->m_NumStride = sizeof(T);
623  return Ptr;
624  }

708  template<typename T>
709  void UpdateVirtexBuffer(const vector<T>& NewBuffer) {
710  auto Ptr = dynamic_pointer_cast< BackupData<T> >(m_BackUpData);
711  if (!Ptr) {
712  throw BaseException(
713  L"バックアップをT型にキャストできません",
714  Util::GetWSTypeName<T>(),
715  L"MeshResource::UpdateVirtexBuffer<T>()"
716  );
717  }
718 
719  if (NewBuffer.size() != Ptr->m_Vertices.size()) {
720  // Map失敗
721  throw BaseException(
722  L"変更する頂点の数がバックアップと違います",
723  L"if (NewBuffer.size() != Ptr->m_Vertices.size())",
724  L"MeshResource::UpdateVirtexBuffer<T>()"
725  );
726  }
727  //座標を変更する
728  auto Dev = App::GetApp()->GetDeviceResources();
729  auto pID3D11DeviceContext = Dev->GetD3DDeviceContext();
730  //頂点バッファをリソースから取り出す
731  auto pVertexBuffer = GetVertexBuffer().Get();
732 
733  //D3D11_MAP_WRITE_DISCARDは重要。この処理により、GPUに邪魔されない
734  D3D11_MAP mapType = D3D11_MAP_WRITE_DISCARD;
735  D3D11_MAPPED_SUBRESOURCE mappedBuffer;
736  //頂点のマップ
737  if (FAILED(pID3D11DeviceContext->Map(pVertexBuffer, 0, mapType, 0, &mappedBuffer))) {
738  // Map失敗
739  throw BaseException(
740  L"頂点のMapに失敗しました。",
741  L"if(FAILED(pID3D11DeviceContext->Map()))",
742  L"MeshResource::UpdateVirtexBuffer<T>()"
743  );
744  }
745  //頂点の変更
746  T* vertices = (T*)mappedBuffer.pData;
747  for (size_t i = 0; i < NewBuffer.size(); i++) {
748  vertices[i] = NewBuffer[i];
749  }
750  //アンマップ
751  pID3D11DeviceContext->Unmap(pVertexBuffer, 0);
752 
753  }
