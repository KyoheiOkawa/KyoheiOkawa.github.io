1183  void Vector3::WorldToSCreen(const Matrix4X4& m, float ViewWidth, float ViewHeight) {
1184  Vector4 Pos4 = *this;
1185  Pos4.w = 1.0f;
1186  //座標変換
1187  Pos4.Transform(m);
1188  //遠近
1189  Pos4.x /= Pos4.w;
1190  Pos4.y /= Pos4.w;
1191  Pos4.z /= Pos4.w;
1192  //座標単位の修正
1193  Pos4.x += 1.0f;
1194  Pos4.y += 1.0f;
1195  Pos4.y = 2.0f - Pos4.y;
1196  //ビューポート変換
1197  Pos4.x *= (ViewWidth * 0.5f);
1198  Pos4.y *= (ViewHeight * 0.5f);
1199  *this = XMVECTOR(Pos4);
1200  }
